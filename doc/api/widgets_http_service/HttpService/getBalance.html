  var _body = jsonDecode(_httpResponse.body);

  if (_parts.length == 1) {
    _whole = _parts[0];
    _decimal = "00";
  } else {
    _whole = _parts[0];

  }
  _whole = _whole.replaceAllMapped(_reg, (match) => '${match[1]},');
